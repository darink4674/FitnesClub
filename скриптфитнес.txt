CREATE VIEW [dbo].[vw_ActiveMemberships] AS
SELECT 
    cm.Id,
    c.Id AS ClientId,
    c.LastName + ' ' + c.FirstName AS ClientName,
    c.Phone,
    mt.Name AS MembershipName,
    cm.StartDate,
    cm.EndDate,
    cm.RemainingVisits,
    DATEDIFF(day, GETDATE(), cm.EndDate) AS DaysRemaining
FROM ClientMemberships cm
INNER JOIN Clients c ON cm.ClientId = c.Id
INNER JOIN MembershipTypes mt ON cm.MembershipTypeId = mt.Id
WHERE cm.Status = 'Active' 
  AND cm.StartDate <= CAST(GETDATE() AS DATE)
  AND cm.EndDate >= CAST(GETDATE() AS DATE);

CREATE VIEW [dbo].[vw_DailyRevenue] AS
SELECT 
    CAST(PaymentDate AS DATE) AS PaymentDay,
    PaymentMethod,
    COUNT(*) AS TransactionsCount,
    SUM(Amount) AS TotalAmount
FROM Payments
WHERE PaymentStatus = 'Completed'
GROUP BY CAST(PaymentDate AS DATE), PaymentMethod;

GO
CREATE VIEW [dbo].[vw_TrainerSchedule] AS
SELECT 
    e.LastName + ' ' + e.FirstName AS TrainerName,
    t.TrainerType,
    t.Specialization,
    ts.DayOfWeek,
    CASE ts.DayOfWeek
        WHEN 1 THEN 'Понедельник'
        WHEN 2 THEN 'Вторник'
        WHEN 3 THEN 'Среда'
        WHEN 4 THEN 'Четверг'
        WHEN 5 THEN 'Пятница'
        WHEN 6 THEN 'Суббота'
        WHEN 7 THEN 'Воскресенье'
    END AS DayName,
    ts.WorkStartTime,
    ts.WorkEndTime,
    ts.BreakStartTime,
    ts.BreakEndTime,
    ts.IsWorkingDay
FROM TrainerSchedule ts
INNER JOIN Trainers t ON ts.TrainerId = t.Id
INNER JOIN Employees e ON t.EmployeeId = e.Id
WHERE (ts.ValidTo IS NULL OR ts.ValidTo >= CAST(GETDATE() AS DATE))
  AND ts.ValidFrom <= CAST(GETDATE() AS DATE);
